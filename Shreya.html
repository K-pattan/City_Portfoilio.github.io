Dijkstra's Algorithm – Shortest Path

Used for: Tourism routes, Industrial supply paths, Solar power routing.


function dijkstra(graph, start) {
  const distances = {};
  const visited = new Set();
  const pq = new MinPriorityQueue();
  for (let node in graph) distances[node] = Infinity;
  distances[start] = 0;
  pq.enqueue(start, 0);

  while (!pq.isEmpty()) {
    let {element: node} = pq.dequeue();
    visited.add(node);
    for (let neighbor in graph[node]) {
      if (!visited.has(neighbor)) {
        let newDist = distances[node] + graph[node][neighbor];
        if (newDist < distances[neighbor]) {
          distances[neighbor] = newDist;
          pq.enqueue(neighbor, newDist);
        }
      }
    }
  }
  return distances;
}

Breadth-First Search (BFS)

Used for: Finding nearby attractions, government service coverage.


function bfs(graph, start) {
  const visited = new Set();
  const queue = [start];
  const result = [];

  while (queue.length) {
    const node = queue.shift();
    if (!visited.has(node)) {
      visited.add(node);
      result.push(node);
      queue.push(...graph[node].filter(n => !visited.has(n)));
    }
  }
  return result;
}

Prim's Algorithm – Minimum Spanning Tree

Used for: Industrial connections, cost optimization.


function primMST(graph) {
  const nodes = Object.keys(graph);
  const visited = new Set();
  const mst = [];
  
  visited.add(nodes[0]);
  while (visited.size < nodes.length) {
    let minEdge = null;
    for (let u of visited) {
      for (let v in graph[u]) {
        if (!visited.has(v)) {
          if (!minEdge || graph[u][v] < minEdge.weight) {
            minEdge = {u, v, weight: graph[u][v]};
          }
        }
      }
    }
    mst.push(minEdge);
    visited.add(minEdge.v);
  }
  return mst;
}

Greedy Algorithm

Used for: Waste truck routing, solar panel placement.


function greedyJobs(jobs) {
  jobs.sort((a,b) => b.profit - a.profit);
  const result = [];
  const slots = Array(jobs.length).fill(false);

  for (let job of jobs) {
    for (let j = Math.min(slots.length-1, job.deadline-1); j>=0; j--) {
      if (!slots[j]) {
        slots[j] = true;
        result[j] = job;
        break;
      }
    }
  }
  return result.filter(Boolean);
}

Queue

Used for: Scheduling waste collection and citizen requests.


class Queue {
  constructor() { this.items = []; }
  enqueue(item) { this.items.push(item); }
  dequeue() { return this.items.shift(); }
  isEmpty() { return this.items.length === 0; }
}
