function dijkstra(graph, start) {
  const dist = {};
  const visited = {};
  const pq = new PriorityQueue();
  for (let node in graph) dist[node] = Infinity;
  dist[start] = 0;
  pq.enqueue([start, 0]);
  while (!pq.isEmpty()) {
    let [node, cost] = pq.dequeue();
    if (visited[node]) continue;
    visited[node] = true;
    for (let neighbor in graph[node]) {
      let newDist = cost + graph[node][neighbor];
      if (newDist < dist[neighbor]) {
        dist[neighbor] = newDist;
        pq.enqueue([neighbor, newDist]);
      }
    }
  }
  return dist;
}

function primMST(graph) {
  const visited = new Set();
  const edges = [];
  const pq = new PriorityQueue();
  let start = Object.keys(graph)[0];
  visited.add(start);
  for(let neighbor in graph[start]){
    pq.enqueue([start, neighbor, graph[start][neighbor]]);
  }
  while(!pq.isEmpty()){
    let [u,v,w] = pq.dequeue();
    if(visited.has(v)) continue;
    visited.add(v);
    edges.push([u,v,w]);
    for(let neighbor in graph[v]){
      if(!visited.has(neighbor)) pq.enqueue([v, neighbor, graph[v][neighbor]]);
    }
  }
  return edges;
}
    
class Queue {
  constructor(){ this.items = []; }
  enqueue(item){ this.items.push(item); }
  dequeue(){ return this.items.shift(); }
  isEmpty(){ return this.items.length === 0; }
}

class HashTable {
  constructor(size=50){ this.table = new Array(size); }
  hash(key){ return key.toString().length % this.table.length; }
  set(key, value){ this.table[this.hash(key)] = value; }
  get(key){ return this.table[this.hash(key)]; }
}
// Example usage: fast lookup of citizen records
let ht = new HashTable();
ht.set("123","John Doe");
